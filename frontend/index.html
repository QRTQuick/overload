Installation & Usage

Python Client

# Install the client library
pip install overload

# Use in your Python code
from overload import analyze

bugs = analyze("your_code.py")
for bug in bugs:
    print(f"{bug['severity']}: {bug['description']}")

REST API

# Direct API usage
curl -X POST https://overload-api.onrender.com/analyze \
  -H "Content-Type: application/json" \
  -d '{"code": "def hello(): print(\"world\")"}'
